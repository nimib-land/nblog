Solution for Combinations_and_permutations

proc perm(n, k: int32): BigInt =
  result = initBigInt 1      ## I could also use `one` (a const)
  var
    k = n - k
    n = n
  while n > k:
    result *= n
    dec n

proc comb(n, k: int32): BigInt =
  result = perm(n, k)
  var k = k
  while k > 0:
    result = result div k
    dec k
